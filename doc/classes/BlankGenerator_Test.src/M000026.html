  def test__generate_by_id_int
    # Test using default values for id_int_base and id_int_step, 0 and 1 respectively. So default for 3 records will be [0, 1, 2]
    dist = @g.generate_by_id_int(3)
    assert(dist == [0, 1, 2], "test__generate_by_id_int failed. dist returned == #{dist}")
    # Test using id_int_base
    dist = @g.generate_by_id_int(3, 10)
    assert(dist == [10, 11, 12], "test__generate_by_id_int failed. dist returned == #{dist}")
    # Test using id_int_base and id_int_step
    dist = @g.generate_by_id_int(3, 10, 2)
    assert(dist == [10, 12, 14], "test__generate_by_id_int failed. dist returned == #{dist}")
  end
