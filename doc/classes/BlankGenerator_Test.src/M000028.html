  def test__generate_by_id_mongo_objid
    require 'mongo'
    dist = @g.generate_by_id_mongo_objid(3)
    assert(dist.length == 3, "generate_by_id_mongo_objid failed. Expected 3 records returned but returned #{dist.length}")
    dist.each do |id|
      assert(BSON::ObjectId.legal?(id.to_s), "generate_by_id_mongo_objid failed. dist returned illegal ObjectId value #{id}")      
      # Based on the source code in the documentation, parse will raise an exception if it can't parse its argument
      assert_nothing_raised do
        BSON::ObjectId.from_string(id.to_s)
      end
    end
  end
