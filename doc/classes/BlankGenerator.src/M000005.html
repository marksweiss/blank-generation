  def generate_field_data(num_records, dist_type, field_def)
    case dist_type
      when "normal"
        generate_by_normal(num_records, field_def['example_value'], field_def['mean'], field_def['std_dev'], field_def['nil_ratio'], field_def['nil_value'])
      when "random"
        generate_by_random(num_records, field_def['example_value'], field_def['min'], field_def['max'], field_def['allow_repeat_values'], field_def['nil_ratio'], field_def['nil_value'])
      when "histogram"
        generate_by_histogram(num_records, field_def['histogram'], field_def['example_value'])
      when "value_set"
        generate_by_value_set(num_records, field_def['value_set'], field_def['nil_ratio'], field_def['nil_value'])
      when "id_int"
        generate_by_id_int(num_records, field_def['id_int_base'], field_def['id_int_step'])
      when "id_uuid"
        generate_by_id_uuid(num_records)
      when "id_mongo_objid"
        generate_by_id_mongo_objid(num_recordse)
      else
        raise BlankGenerationException, "Illegal distribution type #{dist_type} in field definition. See documentation for legal field groupings." 
    end
  end
