  def generate_by_normal(num_records, example_value, mean, std_dev, nil_ratio=nil, nil_value=nil)    
    dist = @r.rnorm(num_records, mean, std_dev)  # rsruby R call
    if example_value.class == Fixnum
      dist.map! {|x| x.floor}
    end
    dist = set_nil_values(dist, nil_ratio, nil_value)
    dist
  end
