  def generate_by_random(num_records, example_value, min, max, allow_repeat_values, nil_ratio=nil, nil_value=nil)    
    dist = []
    if example_value.class == Fixnum
      dist = @r.sample(min.upto(max).collect, num_records, allow_repeat_values)  # rsruby R call, allow_repeat_values allows values in range to be used more than once
    else
      num_records.times do |i|
        dist << rand
      end
    end
    dist = set_nil_values(dist, nil_ratio, nil_value)
    dist
  end
