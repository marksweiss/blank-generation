  def validate(num_records, field_defs, out_format)  
    if num_records.nil? || num_records.class != Fixnum
      raise BlankGenerationException, "Illegal value for num_records #{num_records}"    
    end
  
    if field_defs.class != String && field_defs.class != Array
      raise BlankGenerationException, "Illegal value for field_defs #{field_defs}"    
    end
    
    if out_format != :json && out_format != :csv
      raise BlankGenerationException, "Illegal output format #{out_format}"
    end
  end
