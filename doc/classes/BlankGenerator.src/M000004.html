  def load_field_defs(field_defs)
    case field_defs    
      when String
        require 'json'
        file = File.open(field_defs)
        field_defs = JSON.parse(file.read)
        file.close
      when Hash
        # no-op, already a map
      else
        # no-op, validated in #validate() call above
    end
  
    field_defs 
  end
