  def test__generate_normal
    require 'json'
    
    num_records = 2
    field_name = "test_normal_field"
    field_def = JSON.parse '{"dist_type" : "normal", "example_value" : 1.0, "mean" : 1.0, "std_dev" : 0.0}'
    field_def["name"] = field_name 
    field_defs = [field_def]
    out_format = :json
    dist = @g.generate(num_records, field_defs, out_format)    
    dist = JSON.parse(dist)
    assert(dist.length == num_records, "test__generate failed. dist returned == #{dist.length} records, expected #{num_records}")
    # Test value for each record, each "row", these are the elements in the dist array in sequence    
    record_value = dist[0][field_name]
    assert(record_value == 1.0, "test__generate failed. dist returned == #{record_value}")  
    assert(record_value.class == Float, "test__generate failed. dist returned data of type == #{record_value.class}")  
    record_value = dist[1][field_name]
    assert(record_value == 1.0, "test__generate failed. dist returned == #{record_value}")    
    
    num_records = 2
    field_name = "test_normal_field"
    field_def = JSON.parse '{"dist_type" : "normal", "example_value" : 1, "mean" : 1.0, "std_dev" : 0.0}'
    field_def["name"] = field_name 
    field_defs = [field_def]
    out_format = :json
    dist = @g.generate(num_records, field_defs, out_format)    
    dist = JSON.parse(dist)
    assert(dist.length == num_records, "test__generate failed. dist returned == #{dist.length} records, expected #{num_records}")
    # Test value for each record, each "row", these are the elements in the dist array in sequence    
    record_value = dist[0][field_name]
    assert(record_value == 1, "test__generate failed. dist returned == #{record_value}")    
    assert(record_value.class == Fixnum, "test__generate failed. dist returned data of type == #{record_value.class}")  
    record_value = dist[1][field_name]
    assert(record_value == 1, "test__generate failed. dist returned == #{record_value}")        
  end
