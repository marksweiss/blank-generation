  def test__generate_by_histogram
    # Test for histogram entries providing values
    dist = @g.generate_by_histogram(2, [{"min" => 0, "max" => 0, "count" => 1}])
    assert(dist == [0, 0], "test__generate_by_histogram failed. dist returned == #{dist}")
    assert(dist[0].class == Fixnum, "test__generate_by_histogram returned elements of incorrect type. Expected Fixnum. dist returned == #{dist}")
    assert(dist[1].class == Fixnum, "test__generate_by_histogram returned elements of incorrect type. Expected Fixnum. dist returned == #{dist}")    
    # Test for value_set providing values
    dist = @g.generate_by_histogram(2, [{"value_set" => [10], "count" => 1}])
    assert(dist == [10, 10], "test__generate_by_histogram failed. dist returned == #{dist}")
    dist = @g.generate_by_histogram(2, [{"value_set" => [10, 20], "count" => 1}])
    assert(((dist[0] == 10 or dist[0] == 20) and (dist[1] == 10 or dist[1] == 20)), "test__generate_by_histogram failed. dist returned == #{dist}")
    # Test for dictionary providing values
    dist = @g.generate_by_histogram(2, [{"dictionary" => "test_dictionary.txt", "count" => 1}])    
    assert(((dist[0] == "hello world" or dist[0] == "goodbye chicken") and (dist[1] == "hello world" or dist[1] == "goodbye chicken")), "test__generate_by_histogram failed. dist returned == #{dist}")
  end
