  def test__generate_by_random
    num_records = 2
    example_value = 1
    min = 1
    max = 1
    allow_repeat_values = true
    dist = @g.generate_by_random(num_records, example_value, min, max, allow_repeat_values)
    assert(dist == [1, 1], "test__generate_by_random failed. dist returned == #{dist}")
    max = 2
    dist = @g.generate_by_random(num_records, example_value, min, max, allow_repeat_values)
    assert(((dist[0] == 1 or dist[0] == 2) and (dist[1] == 1 or dist[1] == 2)), "test__generate_by_random failed. dist returned == #{dist}") 
  end
